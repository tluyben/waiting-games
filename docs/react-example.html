2. Loading Page with Controls

Interactive configuration and manual controls:

import React, { useState } from 'react'; import { WaitingGame } from 'waiting-games'; export function LoadingPage() { const [gameConfig, setGameConfig] = useState({ useKeyboard: true, useMobile: true, width: 400, height: 300, theme: 'classic' }); return ( <div style={{ textAlign: 'center', padding: '40px', fontFamily: 'Arial, sans-serif' }}> <h1>‚è≥ Loading Your Data...</h1> <div style={{ background: '#f8f9fa', padding: '20px', borderRadius: '8px', margin: '20px auto', maxWidth: '600px' }}> <p>This might take a minute. Want to play while you wait?</p> <div style={{ margin: '20px 0' }}> <label style={{ marginRight: '20px' }}> <input type="checkbox" checked={gameConfig.useKeyboard} onChange={(e) => setGameConfig({ ...gameConfig, useKeyboard: e.target.checked })} /> Keyboard Controls </label> <label> <input type="checkbox" checked={gameConfig.useMobile} onChange={(e) => setGameConfig({ ...gameConfig, useMobile: e.target.checked })} /> Touch Controls </label> </div> <WaitingGame game="snake" config={gameConfig} autoStart={false} style={{ margin: '20px auto', border: '2px solid #007bff', borderRadius: '8px', display: 'inline-block' }} /> </div> </div> ); }

4. Multiple Games Selector

Game selection with different configurations:

import React, { useState } from 'react'; import { WaitingGame } from 'waiting-games'; export function GameSelector() { const [selectedGame, setSelectedGame] = useState('snake'); const games = [ { id: 'snake', name: 'Snake', width: 400, height: 300 }, { id: 'pong', name: 'Pong', width: 400, height: 300 }, { id: 'tetris', name: 'Tetris', width: 240, height: 420 }, { id: 'pacman', name: 'Pac-Man', width: 392, height: 308 }, { id: 'asteroids', name: 'Asteroids', width: 400, height: 300 }, { id: 'frogger', name: 'Frogger', width: 416, height: 448 } ]; const currentGame = games.find(g => g.id === selectedGame); return ( <div style={{ textAlign: 'center', padding: '20px' }}> <h1>üéÆ Game Selection</h1> <div style={{ margin: '20px 0' }}> <label>Choose a game: </label> <select value={selectedGame} onChange={(e) => setSelectedGame(e.target.value)} style={{ padding: '8px', margin: '0 10px' }} > {games.map(game => ( <option key={game.id} value={game.id}> {game.name} </option> ))} </select> </div> <WaitingGame key={selectedGame} // Force re-render when game changes game={selectedGame} config={{ useKeyboard: true, useMobile: true, width: currentGame.width, height: currentGame.height }} autoStart={true} style={{ margin: '20px auto', border: '2px solid #333', borderRadius: '8px', display: 'inline-block' }} /> </div> ); }

6. Error Boundary Integration

Show games when your app crashes:

import React from 'react'; import { WaitingGame } from 'waiting-games'; class ErrorBoundary extends React.Component { constructor(props) { super(props); this.state = { hasError: false }; } static getDerivedStateFromError(error) { return { hasError: true }; } componentDidCatch(error, errorInfo) { console.error('Error caught by boundary:', error, errorInfo); } render() { if (this.state.hasError) { return ( <div style={{ textAlign: 'center', padding: '40px', fontFamily: 'Arial, sans-serif' }}> <h1>‚ö†Ô∏è Something went wrong</h1> <p>Don't worry, we're fixing it! Play a game while you wait:</p> <WaitingGame game="tetris" config={{ useKeyboard: true, useMobile: true, width: 240, height: 420 }} autoStart={true} /> <button onClick={() => window.location.reload()} style={{ marginTop: '20px', padding: '10px 20px', fontSize: '16px' }} > Reload Page </button> </div> ); } return this.props.children; } } // Usage function App() { return ( <ErrorBoundary> <YourAppComponents /> </ErrorBoundary> ); }
